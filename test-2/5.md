#### ข้อ 5.6
```sql
SELECT *
FROM staff s
INNER JOIN branch b
    ON b.id = s.branch_code
WHERE s.name = 'ระยอง'
    AND s.name = 'กรุงเทพ';
```

#### ข้อ 5.7
```sql
SELECT *
FROM staff s
INNER JOIN branch b
    ON b.id = s.branch_code
INNER JOIN position p
    ON p.id = s.position_code
WHERE b.name = 'เลย'
    AND s.name = 'programmer'
```
